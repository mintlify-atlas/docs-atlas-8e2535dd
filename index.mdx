    Get up and running with Vue QR Style in minutes
  </p>
  
  <Steps>
    <Step title="Install the package">
      Install vue-qr-style and its peer dependency using your preferred package manager.
      
      <CodeGroup>
      ```bash npm
      npm install vue-qr-style qr-code-styling
      ```
      ```bash yarn
      yarn add vue-qr-style qr-code-styling
      ```
      ```bash pnpm
      pnpm add vue-qr-style qr-code-styling
      ```
      </CodeGroup>
    </Step>
    
    <Step title="Register the component">
      Import and register the QR code component in your Vue application.
      
      ```vue
      <script setup>
      import { QrCodeGenerator } from 'vue-qr-style';
      </script>
      ```
    </Step>
    
    <Step title="Create your first QR code">
      Use the component with your desired data and customization options.
      
      ```vue
      <template>
        <QrCodeGenerator
          data="https://github.com/NaturalDevCR/vue-qr-style"
          :width="300"
          :height="300"
          :dotsOptions="{ type: 'rounded', color: '#41b883' }"
        />
      </template>
      ```
    </Step>
  </Steps>
</div>
